def write_signals_perso(conn, user_id: str, as_of, df: pd.DataFrame) -> None:
    """Signaux perso d'un user (RLS owner). Remplace ceux du même (user, as_of)."""
    conn.execute("delete from signals_perso where user_id=%s and as_of=%s", (user_id, as_of))
    sql = ("insert into signals_perso (user_id,ticker,as_of,signal,trigger,pct_vs_buy) "
           "values (%(user_id)s,%(ticker)s,%(as_of)s,%(signal)s,%(trigger)s,%(pct_vs_buy)s)")
    payload = [{"user_id": user_id, "ticker": tk, "as_of": as_of,
                "signal": _clean(df.loc[tk, "signal"]), "trigger": _clean(df.loc[tk, "trigger"]),
                "pct_vs_buy": _clean(df.loc[tk, "pct_vs_buy"])}
               for tk in df.index if df.loc[tk, "signal"] is not None and not pd.isna(df.loc[tk, "signal"])]
    if payload:
        with conn.cursor() as cur:
            cur.executemany(sql, payload)
```

import { createServerClient } from "@supabase/ssr";
import { NextResponse, type NextRequest } from "next/server";

export async function middleware(request: NextRequest) {
  let response = NextResponse.next({ request });
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() { return request.cookies.getAll(); },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value, options }) => response.cookies.set(name, value, options));
        },
      },
    },
  );
  const { data: { user } } = await supabase.auth.getUser();
  const path = request.nextUrl.pathname;
  const isPublic = path.startsWith("/login") || path.startsWith("/auth");
  if (!user && !isPublic) {
    return NextResponse.redirect(new URL("/login", request.url));
  }
  return response;
}

export const config = {
  matcher: ["/((?!_next/static|_next/image|favicon.ico).*)"],
};

import { useState } from "react";

import type { SupabaseClient } from "@supabase/supabase-js";
import type { SignalRow, DeltaRow, PositionRow } from "./types";

export async function getLatestSignals(supabase: SupabaseClient): Promise<SignalRow[]> {
  const { data: maxRow } = await supabase.from("signals").select("as_of").order("as_of", { ascending: false }).limit(1).single();
  if (!maxRow) return [];
  const asOf = maxRow.as_of as string;
  const { data: signals } = await supabase.from("signals").select("ticker,signal,trigger").eq("as_of", asOf);
  const { data: scores } = await supabase.from("scores").select("ticker,combined_z,drawdown_756,rsi_14").eq("as_of", asOf);
  const { data: universe } = await supabase.from("universe").select("ticker,sector");
  const sc = new Map((scores ?? []).map((r) => [r.ticker, r]));
  const un = new Map((universe ?? []).map((r) => [r.ticker, r]));
  return (signals ?? []).map((s) => ({
    ticker: s.ticker, signal: s.signal, trigger: s.trigger,
    sector: un.get(s.ticker)?.sector ?? null,
    combined_z: sc.get(s.ticker)?.combined_z ?? null,
    drawdown_756: sc.get(s.ticker)?.drawdown_756 ?? null,
    rsi_14: sc.get(s.ticker)?.rsi_14 ?? null,
  })).sort((a, b) => (b.combined_z ?? -99) - (a.combined_z ?? -99));
}

export async function getDelta(supabase: SupabaseClient): Promise<DeltaRow[]> {
  const { data: days } = await supabase.from("signals").select("as_of").order("as_of", { ascending: false });
  const uniq = [...new Set((days ?? []).map((d) => d.as_of as string))];
  if (uniq.length === 0) return [];
  const [cur, prev] = [uniq[0], uniq[1]];
  const { data: curr } = await supabase.from("signals").select("ticker,signal,trigger").eq("as_of", cur);
  const prevMap = new Map<string, string | null>();
  if (prev) {
    const { data: p } = await supabase.from("signals").select("ticker,signal").eq("as_of", prev);
    (p ?? []).forEach((r) => prevMap.set(r.ticker, r.signal));
  }
  return (curr ?? []).flatMap((c) => {
    const had = prevMap.has(c.ticker);
    if (!had) return [{ ticker: c.ticker, before: null, signal: c.signal, trigger: c.trigger }];
    if (prevMap.get(c.ticker) !== c.signal) return [{ ticker: c.ticker, before: prevMap.get(c.ticker) ?? null, signal: c.signal, trigger: c.trigger }];
    return [];
  });
}

export async function getPositionsWithPerso(supabase: SupabaseClient): Promise<PositionRow[]> {
  const { data: positions } = await supabase.from("positions").select("id,ticker,quantity,buy_price,buy_date");
  const { data: maxRow } = await supabase.from("signals_perso").select("as_of").order("as_of", { ascending: false }).limit(1).single();
  const perso = new Map<string, { signal: string | null; trigger: string | null; pct_vs_buy: number | null }>();
  if (maxRow) {
    const { data: ps } = await supabase.from("signals_perso").select("ticker,signal,trigger,pct_vs_buy").eq("as_of", maxRow.as_of);
    (ps ?? []).forEach((r) => perso.set(r.ticker, r));
  }
  return (positions ?? []).map((p) => ({
    ...p,
    perso_signal: perso.get(p.ticker)?.signal ?? null,
    perso_trigger: perso.get(p.ticker)?.trigger ?? null,
    pct_vs_buy: perso.get(p.ticker)?.pct_vs_buy ?? null,
  }));
}
```

"use server";
import { revalidatePath } from "next/cache";
import { createClient } from "@/lib/supabase/server";

export async function addPosition(formData: FormData) {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return;
  await supabase.from("positions").insert({
    user_id: user.id,
    ticker: String(formData.get("ticker")).toUpperCase().trim(),
    quantity: Number(formData.get("quantity")),
    buy_price: Number(formData.get("buy_price")),
    buy_date: (formData.get("buy_date") as string) || null,
  });
  revalidatePath("/");
}

export async function deletePosition(formData: FormData) {
  const supabase = await createClient();
  await supabase.from("positions").delete().eq("id", String(formData.get("id")));
  revalidatePath("/");
}
